int led = D7;
int in = D2;
int out =  0;

void setup() {
pinMode(in,INPUT_PULLUP);
Serial.begin(115200);
}

void loop() {
if(digitalRead(in)==LOW){
  
  if(out==0){
   analogWrite(led,85);
    Serial.println("led 85");
    out=85;
    delay(1000);
  }else if(out==85){
    analogWrite(led,170);
  Serial.println("led 170");
  out=170;
   delay(1000);
}else if(out==170){
  analogWrite(led,255);
  Serial.println("led 255");
  out=255;
   delay(1000);
}else if(out==255){
 analogWrite(led,0);
  Serial.println("led 0");
  out=0;
   delay(1000);
}
}
}
